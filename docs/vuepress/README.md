### Sample JavaScript Code

```js
const axios = require("axios");
export default {
  data() {
    return {
      items: []
    };
  },
  beforeMount() {
    axios
      .get("https://jsonplaceholder.typicode.com/users")
      .then(res => {
        console.log(res);
        this.$data.items = res.data;
      })
      .catch(err => {
        console.log(err);
      });
  }
};
```

This is a sample script to be inserted at the end of your .md file.
